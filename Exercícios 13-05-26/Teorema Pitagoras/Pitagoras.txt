import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

class Pitagoras extends JFrame {

    JLabel rotulo1, rotulo2, exibir;
    JTextField texto1, texto2;
    JButton calcular;

    public Pitagoras() {
        super("Teorema de Pitágoras");

        Container tela = getContentPane();
        setLayout(null);

        // Labels
        rotulo1 = new JLabel("Cateto a:");
        rotulo2 = new JLabel("Cateto b:");
        exibir = new JLabel("");

        // Campos de texto
        texto1 = new JTextField(5);
        texto2 = new JTextField(5);

        // Botão
        calcular = new JButton("Calcular");

        // Cores
        rotulo1.setForeground(Color.WHITE);
        rotulo2.setForeground(Color.WHITE);
        exibir.setForeground(Color.WHITE);

        calcular.setForeground(Color.PINK);
        calcular.setBackground(Color.WHITE);

        tela.setBackground(Color.PINK);

        // Posicionamento
        rotulo1.setBounds(50, 20, 100, 20);
        texto1.setBounds(150, 20, 150, 20);

        rotulo2.setBounds(50, 60, 100, 20);
        texto2.setBounds(150, 60, 150, 20);

        calcular.setBounds(50, 100, 250, 25);
        exibir.setBounds(50, 140, 300, 20);

        // Evento do botão
        calcular.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                try {
                    double a = Double.parseDouble(texto1.getText());
                    double b = Double.parseDouble(texto2.getText());
                    double c = Math.sqrt(a * a + b * b);

                    exibir.setText("Hipotenusa = " + c);
                    exibir.setVisible(true);
                } catch (NumberFormatException ex) {
                    exibir.setText("Digite apenas números válidos!");
                    exibir.setVisible(true);
                }
            }
        });

        exibir.setVisible(false);

        // Adicionando componentes
        tela.add(rotulo1);
        tela.add(texto1);
        tela.add(rotulo2);
        tela.add(texto2);
        tela.add(calcular);
        tela.add(exibir);

        // Configurações da janela
        setSize(400, 220);
        setVisible(true);
    }

    public static void main(String[] args) {
        Pitagoras app = new Pitagoras();
        app.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }
}
