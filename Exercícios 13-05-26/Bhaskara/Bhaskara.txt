import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

class Bhaskara extends JFrame {

    JLabel rotulo1, rotulo2, rotulo3, exibir;
    JTextField texto1, texto2, texto3;
    JButton calcular;

    public Bhaskara() {
        super("Fórmula de Bhaskara");

        Container tela = getContentPane();
        setLayout(null);

        // Labels
        rotulo1 = new JLabel("Valor de a:");
        rotulo2 = new JLabel("Valor de b:");
        rotulo3 = new JLabel("Valor de c:");
        exibir = new JLabel("");

        // Campos de texto
        texto1 = new JTextField(5);
        texto2 = new JTextField(5);
        texto3 = new JTextField(5);

        // Botão
        calcular = new JButton("Calcular");

        // Cores
        rotulo1.setForeground(Color.WHITE);
        rotulo2.setForeground(Color.WHITE);
        rotulo3.setForeground(Color.WHITE);
        exibir.setForeground(Color.WHITE);

        calcular.setForeground(Color.PINK);
        calcular.setBackground(Color.WHITE);

        tela.setBackground(Color.PINK);

        // Posicionamento (estilo calculadora)
        rotulo1.setBounds(50, 20, 100, 20);
        texto1.setBounds(150, 20, 150, 20);

        rotulo2.setBounds(50, 60, 100, 20);
        texto2.setBounds(150, 60, 150, 20);

        rotulo3.setBounds(50, 100, 100, 20);
        texto3.setBounds(150, 100, 150, 20);

        calcular.setBounds(50, 140, 250, 25);
        exibir.setBounds(50, 180, 300, 40);

        // Evento do botão
        calcular.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                try {
                    double a = Double.parseDouble(texto1.getText());
                    double b = Double.parseDouble(texto2.getText());
                    double c = Double.parseDouble(texto3.getText());

                    double delta = b * b - 4 * a * c;

                    if (delta < 0) {
                        exibir.setText("Delta negativo! Sem raízes reais.");
                    } else {
                        double x1 = (-b + Math.sqrt(delta)) / (2 * a);
                        double x2 = (-b - Math.sqrt(delta)) / (2 * a);
                        exibir.setText("x1 = " + x1 + " | x2 = " + x2);
                    }

                    exibir.setVisible(true);
                } catch (NumberFormatException ex) {
                    exibir.setText("Digite apenas números válidos!");
                    exibir.setVisible(true);
                }
            }
        });

        exibir.setVisible(false);

        // Adicionando componentes
        tela.add(rotulo1);
        tela.add(texto1);
        tela.add(rotulo2);
        tela.add(texto2);
        tela.add(rotulo3);
        tela.add(texto3);
        tela.add(calcular);
        tela.add(exibir);

        setSize(400, 270);
        setVisible(true);
    }

    public static void main(String[] args) {
        Bhaskara app = new Bhaskara();
        app.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }
}
