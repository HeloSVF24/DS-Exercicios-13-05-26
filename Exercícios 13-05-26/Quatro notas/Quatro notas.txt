import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

class QuatroNotas extends JFrame {
    JLabel rotulo1, rotulo2, rotulo3, rotulo4, exibir;
    JTextField nota1, nota2, nota3, nota4;
    JButton calcular;

    public QuatroNotas() {
        super("Calculadora de Média");
        Container tela = getContentPane();
        setLayout(null);

        // Labels
        rotulo1 = new JLabel("1° nota:");
        rotulo2 = new JLabel("2° nota:");
        rotulo3 = new JLabel("3° nota:");
        rotulo4 = new JLabel("4° nota:");
        exibir = new JLabel("");

        // TextFields
        nota1 = new JTextField(5);
        nota2 = new JTextField(5);
        nota3 = new JTextField(5);
        nota4 = new JTextField(5);

        // Botão
        calcular = new JButton("Calcular Média");

        // Cores
        rotulo1.setForeground(Color.WHITE);
        rotulo2.setForeground(Color.WHITE);
        rotulo3.setForeground(Color.WHITE);
        rotulo4.setForeground(Color.WHITE);
        exibir.setForeground(Color.WHITE);
        calcular.setForeground(Color.PINK);
        calcular.setBackground(Color.WHITE);
        tela.setBackground(Color.PINK);

        // Posicionamento
        rotulo1.setBounds(20, 20, 80, 20);
        nota1.setBounds(100, 20, 150, 20);
        rotulo2.setBounds(20, 50, 80, 20);
        nota2.setBounds(100, 50, 150, 20);
        rotulo3.setBounds(20, 80, 80, 20);
        nota3.setBounds(100, 80, 150, 20);
        rotulo4.setBounds(20, 110, 80, 20);
        nota4.setBounds(100, 110, 150, 20);
        calcular.setBounds(20, 150, 230, 25);
        exibir.setBounds(20, 190, 250, 25);

        // Ação do botão
        calcular.addActionListener(
            new ActionListener() {
                public void actionPerformed(ActionEvent e) {
                    try {
                        double n1 = Double.parseDouble(nota1.getText());
                        double n2 = Double.parseDouble(nota2.getText());
                        double n3 = Double.parseDouble(nota3.getText());
                        double n4 = Double.parseDouble(nota4.getText());
                        double media = (n1 + n2 + n3 + n4) / 4;
                        exibir.setText("A média é: " + media);
                        exibir.setVisible(true);
                    } catch (NumberFormatException ex) {
                        exibir.setText("Digite apenas números válidos!");
                        exibir.setVisible(true);
                    }
                }
            }
        );

        // Adicionando à tela
        tela.add(rotulo1);
        tela.add(nota1);
        tela.add(rotulo2);
        tela.add(nota2);
        tela.add(rotulo3);
        tela.add(nota3);
        tela.add(rotulo4);
        tela.add(nota4);
        tela.add(calcular);
        tela.add(exibir);

        setSize(300, 270);
        setVisible(true);
    }

    public static void main(String[] args) {
        QuatroNotas app = new QuatroNotas();
        app.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }
}
